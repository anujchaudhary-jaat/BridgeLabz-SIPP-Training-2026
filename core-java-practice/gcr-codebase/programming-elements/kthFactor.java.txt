class kthFactor {
    public int kthFactors(int n, int k) {
        int j=0;
        int l=0;
        for(int i=1;i<=n;i++){
            if(n%i==0){
                j++;
            }
        }
        if(k>j){
            return -1;
        }
        int[] arr=new int[j];
        for(int m=1;m<=n;m++){
            if(n%m==0){
                arr[l]=m;
                l++;
            }
        }
        return arr[k-1];
    
    }
}
