public class StringValidator {

    public static boolean isValidUsername(String username) {
        // Base case: empty string is valid
        if (username.length() == 0) {
            return true;
        }

        char ch = username.charAt(0);

        // Check if first character is a lowercase letter
        if (ch < 'a' || ch > 'z') {
            return false;
        }

        // Recursive call on the remaining substring
        return isValidUsername(username.substring(1));
    }

    public static void main(String[] args) {
        System.out.println(isValidUsername("abcdxyz")); // true
        System.out.println(isValidUsername("abcD123")); // false
    }
}
