public class EmployeeSkillAssignment {

    public static void findTeams(int[] skills, int target) {
        findTeams(skills, target, 0, "");
    }

    private static void findTeams(int[] skills, int target, int index, String team) {
        // If target is achieved, print the team
        if (target == 0) {
            System.out.println("[" + team.trim() + "]");
            return;
        }

        // Base case
        if (index == skills.length || target < 0) {
            return;
        }

        // Include current employee
        findTeams(skills, target - skills[index],
                  index + 1,
                  team + skills[index] + " ");

        // Exclude current employee
        findTeams(skills, target,
                  index + 1,
                  team);
    }

    public static void main(String[] args) {
        int[] skills = {2, 3, 5, 7};
        int target = 10;

        findTeams(skills, target);
    }
}
